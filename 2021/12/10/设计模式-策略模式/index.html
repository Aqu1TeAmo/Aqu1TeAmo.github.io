设计模式--策略模式

发表于 2021-12-10 分类于 技术学习

策略模式(Stragegy Pattern)

定义一系列算法，将每一个算法封装起来，并让它们可以相互替换。策略模式让算法独立于使用它的客户而变化。

优点：
1、算法可以自由切换。
2、避免使用多重条件判断。
3、扩展性良好。

缺点：
1、策略类会增多。
2、所有策略类都需要对外暴露。

示例，以商场收银小程序为例:

package strategy

//CashSuper 现金收费接口
type CashSuper interface {
    AcceptCash(money float64) float64
}

//CashNormal 正常收费
type CashNormal struct { 

}

//AcceptCash 正常收费
func (*CashNormal) AcceptCash(money float64) float64 {
    return money
}

//CashRebate 打折收费
type CashRebate struct {
    moneyRebate float64
}

//NewCashRebate 构造函数
func NewCashRebate(moneyRebate float64) CashSuper {
    return &CashRebate{
        moneyRebate: moneyRebate,
    }
}

//AcceptCash 打折收费 乘以折扣
func (rebate *CashRebate) AcceptCash(money float64) float64 {
    return money * rebate.moneyRebate
}

//cashReturn 返利收费
type CashReturn struct {
    moneyCondition, moneyReturn float64
}

//NewCashReturn 构造函数，返回CashReturn
func NewCashReturn(moneyCondition, moneyReturn float64) CashSuper {
    return &CashReturn{
        moneyCondition: moneyCondition,
        moneyReturn:    moneyReturn,
    }
}

//AcceptCash 返利收费
func (re *CashReturn) AcceptCash(money float64) float64 {
    if money >= re.moneyCondition {
        money -= re.moneyReturn
    }
    return money
}

//CashContext 收费策略
type CashContext struct {
    cs CashSuper
}

//NewCashContext CashContext构造函数
func NewCashContext(cs CashSuper) *CashContext {
    return &CashContext{
        cs: cs,
    }
}

func (context *CashContext) GetResult(money float64) float64 {
    return context.cs.AcceptCash(money)
}

测试代码:

package strategy

import "testing"

func TestStrategy(t *testing.T){
    var cc *CashContext
    strType :="打八折"
    switch strType{
    case "正常收费":
        cc = NewCashContext(new(CashNormal))
    case "满300返100":
        cc = NewCashContext(NewCashReturn(300,100))
    case "打八折":
        cc = NewCashContext(NewCashRebate(0.8))
    }
    res :=cc.GetResult(300)
    if res != 240 {
        t.Fatal("CashRebate test fail")
    }
}
